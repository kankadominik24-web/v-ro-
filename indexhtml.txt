        <p id="errorMsg" class="error">Zkus to znovu, tohle není ono.</p>
        
        <button onclick="checkWord()">Ověřit</button>

    <script>
        function checkWord() {
            const input = document.getElementById('wordInput');
            const result = document.getElementById('result');
            const error = document.getElementById('errorMsg');
            
            // Převede vstup na malá písmena a odstraní mezery
            const value = input.value.trim().toLowerCase();

            if (value === "heslo") {
                // Skryje chybu a ukáže úspěch
                error.style.display = 'none';
                result.style.display = 'block';
                input.style.borderColor = '#28a745';
                
                // Volitelné: znemožní další psaní po úspěchu
                input.disabled = true;
                document.querySelector('button').disabled = true;
                document.querySelector('button').style.opacity = '0.5';
            } else {
                // Ukáže chybu
                error.style.display = 'block';
                input.style.borderColor = '#dc3545';
                result.style.display = 'none';
            }
        }

        // Umožní odeslání stiskem klávesy Enter
        document.getElementById('wordInput').addEventListener('keypress', function (e) {
            if (e.key === 'Enter') {
                checkWord();
            }
        });
    </script>
